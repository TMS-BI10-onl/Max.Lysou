--Задача 1
--Выберите из таблицы Водители записи, где стаж водителя не равен 3, 5, 6, 10 и  место прописки 'eee'.

SELECT *
FROM БД.Водители
WHERE Стаж NOT IN  (3, 5, 6, 10) AND Место_прописки LIKE 'eee';


--Задача 2
--Найдите № паспортов водителей, которые ни разу не совершали поездки (решите задачу через left join).

SELECT 
	ВОД.№_паспорта
FROM БД.Водители ВОД
	LEFT JOIN БД.Диспетчерская ДИС
	ON ВОД.Код_водителя = ДИС.Код_водителя
WHERE ДИС.Код_водителя IS NULL;


--Задача 3
--Написать запрос на выборку авто с годом выпуска = 2015 и его маршруты (таблицы Aвто, Диспеческая, Маршрут). 
--Вывести рег номер авто, время прибытия и отбытия, пункт назначения.

SELECT 
	АВТ.Регистрационный_№_авто,
	ДИС.Время_прибытия,
	ДИС.Время_отбытия,
	МАР.Пункт_назначения
FROM БД.Автомобили АВТ
	LEFT JOIN БД.Диспетчерская ДИС
	ON АВТ.Код_авто = ДИС.Код_авто
	LEFT JOIN БД.Маршрут МАР
--	ON ДИС.Код_маршрута = МАР.Код_маршрута
--Если Путевка содержит ключ для соединения с таблицей Маршруты
	ON ДИС.Путевка = МАР.Код_маршрута
WHERE АВТ.Год_выпуска_авто = 2015;
